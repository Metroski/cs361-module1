// Task 4: Split into Modules
import { Product } from "./product.js";

const statusMessage = document.querySelector("#status-message");
const productList = document.querySelector("#product-list");
const cartCountEl = document.querySelector("#cart-count");
const cartTotalEl = document.querySelector("#cart-total");

// Task 5: A Persistent Cart
function loadCart() {
    const stored = localStorage.getItem("cart");
    return stored ? JSON.parse(stored) : [];
}

function saveCart(cart) {
    localStorage.setItem("cart", JSON.stringify(cart));
}

let cart = loadCart();

function renderCart() {
    const total = cart.reduce((sum, item) => sum + item.price, 0);
    cartCountEl.textContent = cart.length;
    cartTotalEl.textContent = total.toFixed(2);
}

function addToCart(product) {
    cart.push({ name: product.name, price: product.withTax() });
    saveCart(cart);
    renderCart();
}

// Task 1 & 6: Fetch & Display, Bring It Together
async function loadProducts() {
    try {
        statusMessage.textContent = "Loading...";

        const response = await fetch("https://jsonplaceholder.typicode.com/users");

        if (!response.ok) {
            throw new Error("Network response was not ok");
        }

        const users = await response.json();

        // Turn each fetched user into a Product so we can reuse the class
        const products = users.map((user, index) => new Product(user.name, (index + 1) * 15));

        statusMessage.textContent = "";
        renderProducts(products, users);
    } catch (error) {
        // Task 2: Loading & Error States
        statusMessage.textContent = "Something went wrong loading products. Please try again later.";
        console.error(error);
    }
}

function renderProducts(products, users) {
    productList.innerHTML = products
        .map((product, index) => {
            const user = users[index];
            return `
                <div class="product-card" data-index="${index}">
                    <h3>${product.name}</h3>
                    <p>${user.email}</p>
                    <p>Price (incl. tax): $${product.withTax().toFixed(2)}</p>
                    <button class="add-to-cart-btn" data-index="${index}">Add to Cart</button>
                </div>
            `;
        })
        .join("");

    document.querySelectorAll(".add-to-cart-btn").forEach((btn) => {
        btn.addEventListener("click", () => {
            const index = Number(btn.dataset.index);
            addToCart(products[index]);
        });
    });
}

renderCart();
loadProducts();
